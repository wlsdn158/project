git
git --version //version 확인

cd C:\test
git clone https://github.com/wlsdn158/Git-First//깃 주소에 있는 파일을 불러온다 clone은 다운로드라고 생각한다
cd Git-First  //깃에서 만든 파일 이름
git add doc.txt //git에 doc 파일을 추가
git commit -m "Add Text File [document.txt]"  //commit은 나중에 오류생겼을때 복구 목적으로도 쓰임
git push // 실제 깃 주소에도 파일을 보냄
